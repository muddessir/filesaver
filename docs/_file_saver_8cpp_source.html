11 {
12  const std::array<std::string, 7> suffixes{
13  "B",
14  "KB",
15  "MB",
16  "GB",
17  "TB",
18  "PB",
19  "EB",
20  };
21  uint suffixIndex = 0;
22  auto count = static_cast<double> (bytes);
23  while (count >= 1024 && suffixIndex < 7)
24  {
25  suffixIndex++;
26  count /= 1024;
27  }
28

30  {
31  return (boost::format ("%d%s") % count % suffixes[suffixIndex]).str ();

132 void FileSaver::entryReader ()
133 {
134  unsigned long iterations = 0;
135 
136  while (running)
137  {
138  while (manager.resultQueue.size () > 0)
139  {
140  auto entry = manager.resultQueue.front ();
141 
142  // This is made to minimize contention on the lock. Result queue will block waiting on the result queue lock
143  // and on there being no results.
144  {
145  std::unique_lock<std::mutex> lock{criticalSection};
146  auto filepath = entry->filepath;
147  auto filepathStr = entry->filepath.string ();
148 
149  allEntries[filepathStr] = entry;
150 
151  if (!entry->isDirectory ())
152  {
153  pendingChildren[filepathStr] = 0;
154  onFinished (entry->filepath);
155  }
156  else
157  {
158  pendingChildren[filepathStr] = entry->children ().size ();
159  if (pendingChildren[filepathStr] == 0)
160  {
161  onFinished (entry->filepath);
162  }
163  }
164  updateSizes (entry);
165  totalFiles += 1;
166  totalKnownFiles += entry->children ().size ();
167  }
168  }
169 
170  iterations += 1;
171 
172  std::this_thread::sleep_for (std::chrono::milliseconds (300));
173  }
174 }

176 void FileSaver::entryWriter ()
177 {
178  while (running)
179  {
180  {
181  // This is made to minimize contention on the lock. Rather than holding it while writing to the DB,
182  // we'll copy the entries to store onto a buffer then release the lock.
183  std::vector<std::shared_ptr<FileEntry>> entriesToStore;
184  {
185  std::unique_lock<std::mutex> lock{criticalSection};
186  while (storageQueue.size () > 0)
187  {
188  auto entry = storageQueue.front ();
189  entriesToStore.push_back (entry);
190  }
191  }
192 
193  for (auto& entry : entriesToStore)
194  {
195  storageService->insertEntry (*entry);
196  }
197 
198  spdlog::info ("Flushing {0} entries", entriesToStore.size ());
199  }
200 
201  std::this_thread::sleep_for (std::chrono::milliseconds (1000));
202  }

205 void FileSaver::onFinished (const boost::filesystem::path& filepath)
206 {
207  const auto& filepathStr = filepath.string ();
208  auto entry = allEntries[filepathStr];
209  entry->isFinished = true;
210  allEntries.erase (filepathStr);
211 
212  if (hasStorage ())
213  {
214  storageQueue.push (std::move (entry));
215  }
216 
217  if (filepath.has_parent_path ())
218  {
219  auto parentPath = filepath.parent_path ();
220  const auto& parentPathStr = parentPath.string ();
221 
222  pendingChildren[parentPathStr] -= 1;
223  if (pendingChildren[parentPathStr] <= 0)
224  {
225  onFinished (parentPath);
226  }
227  }
